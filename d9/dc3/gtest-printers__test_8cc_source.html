543 // An assertion predicate determining whether a one string is a prefix for
544 // another.
545 template <typename StringType>
546 AssertionResult HasPrefix(const StringType& str, const StringType& prefix) {
547  if (str.find(prefix, 0) == 0)
548  return AssertionSuccess();
549 
550  const bool is_wide_string = sizeof(prefix[0]) > 1;
551  const char* const begin_string_quote = is_wide_string ? "L\"" : "\"";
552  return AssertionFailure()
553  << begin_string_quote << prefix << "\" is not a prefix of "
554  << begin_string_quote << str << "\"\n";
555 }
