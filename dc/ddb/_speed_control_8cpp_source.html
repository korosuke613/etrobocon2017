34 
35  float distance4msFloat = 0.0; //4msの距離
36 
37  // 4ms間の走行距離 = ((円周率 * タイヤの直径) / 360) * (モータ角度過去値　- モータ角度現在値)

40  distance4msFloat = (distance4msL + distance4msR) / 2.0; //左右タイヤの走行距離を足して割る
41

43  preAngleL = curAngleL;
44  preAngleR = curAngleR;
