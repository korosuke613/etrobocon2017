58 void SelfLocalization::update () {
59 
60  //左車輪の回転角
61  left_motor_old_angle = left_motor_current_angle;
62  left_motor_current_angle = ev3_motor_get_counts(left_motor_sl);
63  left_motor_rotation_angle =
64  left_motor_current_angle - left_motor_old_angle;
65  left_wheel_moving_distance =
66  wheel_across * M_PI * (left_motor_rotation_angle / 360.0);
67 
68  //右車輪の回転角
69  old_x = current_x;
70  old_y = current_y;
71  old_angle = current_angle;
72 
73  right_motor_old_angle = right_motor_current_angle;
74  right_motor_current_angle = ev3_motor_get_counts(right_motor_sl);
75  right_motor_rotation_angle =
76  right_motor_current_angle - right_motor_old_angle;
77  right_wheel_moving_distance =
78  wheel_across * M_PI * (right_motor_rotation_angle / 360.0);
79 
80  //移動距離と、車体の回転角
81  moving_distance =
82  (right_wheel_moving_distance + left_wheel_moving_distance) / 2;
83  turning_angle = (right_wheel_moving_distance - left_wheel_moving_distance)
84  / between_wheels;
85 
86  //座標
87  current_x = old_x + (moving_distance * cos(old_angle + (turning_angle/2)));
88  current_y = old_y + (moving_distance * sin(old_angle + (turning_angle/2)));
89  current_angle = old_angle + turning_angle;
90 
91 }
