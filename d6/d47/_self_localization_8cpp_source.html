46 void SelfLocalization::update () {
47 
48  //左車輪の回転角
49  left_motor_old_angle = left_motor_current_angle;
50  left_motor_current_angle = ev3_motor_get_counts(left_motor_sl);
51  left_motor_rotation_angle =
52  left_motor_current_angle - left_motor_old_angle;
53  left_wheel_moving_distance =
54  wheel_across * M_PI * (left_motor_rotation_angle / 360.0);
55 
56  //右車輪の回転角
57  old_x = current_x;
58  old_y = current_y;
59  old_angle = current_angle;
60 
61  right_motor_old_angle = right_motor_current_angle;
62  right_motor_current_angle = ev3_motor_get_counts(right_motor_sl);
63  right_motor_rotation_angle =
64  right_motor_current_angle - right_motor_old_angle;
65  right_wheel_moving_distance =
66  wheel_across * M_PI * (right_motor_rotation_angle / 360.0);
67 
68  //移動距離と、車体の回転角
69  moving_distance =
70  (right_wheel_moving_distance + left_wheel_moving_distance) / 2;
71  turning_angle = (right_wheel_moving_distance - left_wheel_moving_distance)
72  / between_wheels;
73 
74  //座標
75  current_x = old_x + (moving_distance * cos(old_angle + (turning_angle/2)));
76  current_y = old_y + (moving_distance * sin(old_angle + (turning_angle/2)));
77  current_angle = old_angle + turning_angle;
78 
79 }
