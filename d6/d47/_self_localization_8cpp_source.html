29 void SelfLocalization::update (std::int32_t left_motor_sl, std::int32_t right_motor_sl) {
30 
31  //左車輪の回転角
32  left.update(left_motor_sl);
33 
34  //右車輪の回転角
35  right.update(right_motor_sl);
36 
37  //移動距離と、車体の回転角
38  moving_distance_mean = (right.moving_distance + left.moving_distance) / 2;
39  turning_angle = (right.moving_distance - left.moving_distance) / between_wheels;
40 
41  //座標
42  current_x += (moving_distance_mean * std::cos(current_angle + (turning_angle/2)));
43  current_y += (moving_distance_mean * std::sin(current_angle + (turning_angle/2)));
44  current_angle += turning_angle;
45 
46  //保存
47  if(isSave == true){
48  writing_current_coordinates();
49  }
50 }

67 bool SelfLocalization::approached_target_coordinates (float target_x, float target_y, float target_radius) {
68  float distance = std::sqrt(
69  (target_x - current_x) * (target_x - current_x) +
70  (target_y - current_y) * (target_y - current_y) );
71  if(distance < target_radius)
72  return true;
73  return false;
74 }

79 float SelfLocalization::calculate_between_ev3_and_border
80 (float _start_x, float _start_y, float _goal_x, float _goal_y, float _current_x, float _current_y) {
81  float a, b, c;
82 
83  a = _goal_y - _start_y;
84  b = -(_goal_x - _start_x);
85  c = -b * _start_y - a * _start_x;
86 
87  return (a*_current_x + b*_current_y + c) / std::sqrt(a*a + b*b);
88 }

108 bool SelfLocalization::is_over_normal_vector(float _start_x, float _start_y, float _goal_x, float _goal_y, float _current_x, float _current_y) {
109  float k = (_start_x - _goal_x) / (_goal_y - _start_y);
110  float border_y = k * _current_x + _goal_y - k * _goal_x;
111  if(_start_y >= _goal_y )return _current_y >= border_y;
112  return _current_y < border_y;
113 }
