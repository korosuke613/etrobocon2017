35 void SelfLocalization::update (std::int32_t left_motor_sl, std::int32_t right_motor_sl) {
36 
37  //左車輪の回転角
38  left.update(left_motor_sl);
39 
40  //右車輪の回転角
41  right.update(right_motor_sl);
42 
43  //移動距離と、車体の回転角
44  moving_distance_mean = (right.moving_distance + left.moving_distance) / 2;
45  turning_angle = (right.moving_distance - left.moving_distance) / between_wheels;
46 
47  //座標
48  current_x += (moving_distance_mean * cos(current_angle + (turning_angle/2)));
49  current_y += (moving_distance_mean * sin(current_angle + (turning_angle/2)));
50  current_angle += turning_angle;
51 
52  //保存
53  if(isSave == true){
54  writing_current_coordinates();
55  }
56 }

73 bool SelfLocalization::approached_target_coordinates (float target_x, float target_y, float target_radius) {
74  float distance = sqrt(
75  (target_x - current_x) * (target_x - current_x) +
76  (target_y - current_y) * (target_y - current_y) );
77  if(distance < target_radius)
78  return true;
79  return false;
80 }
